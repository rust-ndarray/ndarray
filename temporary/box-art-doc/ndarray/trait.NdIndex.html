[−][src]Trait ndarray::NdIndex

pub unsafe trait NdIndex<E>: Debug { }

Tuple or fixed size arrays that can be used to index an array.

use ndarray::arr2;

let mut a = arr2(&[[0, 1],
                   [2, 3]]);
assert_eq!(a[[0, 1]], 1);
assert_eq!(a[[1, 1]], 3);
a[[1, 1]] += 1;
assert_eq!(a[(1, 1)], 4);
