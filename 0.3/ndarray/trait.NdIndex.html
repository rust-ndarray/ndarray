A tuple or fixed size array that can be used to index an array.

use ndarray::arr2;

let mut a = arr2(&[[0, 1], [0, 0]]);
a[[1, 1]] = 1;
assert_eq!(a[[0, 1]], 1);
assert_eq!(a[[1, 1]], 1);
